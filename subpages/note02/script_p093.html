논리(불) 연산자

Bool 연산자

    let res1 = true && true;
    let res2 = true && false;
    let res3 = false && true;
    let res4 = false && false;
    let res5 = true || true;
    let res6 = true || false;
    let res7 = false || true;
    let res8 = false || false;
    
    console.log(`true && true   : ${res1}`);
    console.log(`true && false  : ${res2}`);
    console.log(`false && true  : ${res3}`);
    console.log(`false && false : ${res4}`);
    console.log(`true || true   : ${res5}`);
    console.log(`true || false  : ${res6}`);
    console.log(`false || true  : ${res7}`);
    console.log(`false || false : ${res8}`);
