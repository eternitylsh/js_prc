    let b1 = !true;
    let b2 = !!false;
    let b3 = 57 > 333;
    let b4 = (234 == '234') // 같음으로..
    let b5 = "사랑" > "하마"

    console.log( `!true           :       ${b1}` );
    console.log( `!!false         :       ${b2}` );
    console.log( `57 > 333        :       ${b3}` );
    console.log( `234 == '234'    :       ${b4}` );
    console.log( `"사랑" > "하마"  :       ${b5}` );
